Proxy implementation 

Operates at layer 7
Browser -> Gateway proxy (NestJS) -> Inspection engine -> Policy engine -> Destination website

Lifecycle:
	1. Incoming request
	GET http://example.com/login
	Controller:
	@Controller('*')
	export class ProxyController {
	  @All()
	  async handle(@Req() req, @Res() res) {
	    return this.proxyService.forward(req, res);
	  }
	}
	2. Identity resolution
	Proxy extracts session: cookie -> session -> user
	3. Build inspection context
	const context = {
	  userId,
	  url,
	  method,
	  headers,
	  timestamp
	};
	4. Inspection engine
	Security checks: URL reputation -> Domain policy -> Heuristics -> Cache lookup
	Service: inspectionService.inspect(context)
	Returns: 
	{
	  verdict: "ALLOW" | "BLOCK" | "WARN",
	  reason: "malware_detected"
	}
	5. Failure behavior
	VirusTotal timeout: Allow + warning
	Inspection crash: Block
	DB unavailable: Block
	6. Policy engine
	User policies applied… blocked domains, time restrictions, risk threshold
	7. Decision
	Block: custom firewall page
	Allow: forward request
	Warn: inject banner
	8. Forward request
	Using Node streaming proxy:
	proxy.web(req, res, {
	  target: targetUrl,
	  changeOrigin: true
	});
	9. Log event
	eventBus.emit("TRAFFIC_LOG", log);
	10. WebSocket broadcast
	Dashboard updates
	
Auth + Vault Crypto Implementation Walkthrough

Server never stores plaintext secrets.
Even server cannot decrypt without password.

Flow:
	1. User registers 
	Provides email and password
	2. Password hash (authentication)
	argon2.hash(password)
	Stored password_hash used only for login verification
	3. Key derivation (Encryption)
	masterKey = KDF(password + userSalt)
	crypto.scryptSync(password, salt, 32); result 32 bytes = AES-256 key
  #   Login auth - argon2 hash
       Encryption  - derived AES key
	4. Vault encryption
	Saving secret: cipher = AES-256-GCM(masterKey)
	Stored in DB: ciphertext, iv, auth_tag
	5. Runtime key handling
	Password -> derive key and keep in memory, never stored, key destroyed after session expiration
	Sensitive action reverification: before viewing secrets, changing policy, exporting logs frontend asks for password again, server derives key again and attempts decrypt test value

//////////////////////////////////////////////////
// USERS
//////////////////////////////////////////////////

model User {
  id            String   @id @default(uuid())
  email         String   @unique
  passwordHash  String
  kdfSalt       String   // used for encryption key derivation
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt

  sessions      Session[]
  vaultItems    VaultItem[]
  policies      Policy[]
  trafficLogs   TrafficLog[]
  auditLogs     AuditLog[]
}

//////////////////////////////////////////////////
// SESSIONS (HttpOnly Cookie Sessions)
//////////////////////////////////////////////////

model Session {
  id          String   @id @default(uuid())
  userId      String
  expiresAt   DateTime
  createdAt   DateTime @default(now())
  lastSeenAt  DateTime

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId])
}

//////////////////////////////////////////////////
// VAULT (Encrypted Secrets)
//////////////////////////////////////////////////

model VaultItem {
  id          String   @id @default(uuid())
  userId      String
  name        String

  ciphertext  Bytes
  iv          Bytes
  authTag     Bytes

  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId])
}

//////////////////////////////////////////////////
// USER SECURITY POLICIES
//////////////////////////////////////////////////

model Policy {
  id          String   @id @default(uuid())
  userId      String

  blockedDomains String[]
  riskThreshold  Int     @default(50)
  enabled        Boolean @default(true)

  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)
}

//////////////////////////////////////////////////
// TRAFFIC LOGS (HIGH VOLUME)
//////////////////////////////////////////////////

model TrafficLog {
  id         String   @id @default(uuid())
  userId     String

  url        String
  method     String
  verdict    Verdict
  reason     String?

  statusCode Int?
  ipAddress  String?

  createdAt  DateTime @default(now())

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId])
  @@index([createdAt])
  @@index([verdict])
}

enum Verdict {
  ALLOW
  BLOCK
  WARN
}

//////////////////////////////////////////////////
// SECURITY / AUDIT EVENTS
//////////////////////////////////////////////////

model AuditLog {
  id        String   @id @default(uuid())
  userId    String?

  eventType AuditEvent
  metadata  Json

  createdAt DateTime @default(now())

  user User? @relation(fields: [userId], references: [id])

  @@index([createdAt])
  @@index([eventType])
}

enum AuditEvent {
  LOGIN_SUCCESS
  LOGIN_FAILURE
  PASSWORD_REVERIFY
  POLICY_CHANGED
  VAULT_ACCESS
  SECURITY_ALERT
}


Example query:
await prisma.trafficLog.findMany({
  where: {
    verdict: "BLOCK"
  },
  orderBy: {
    createdAt: "desc"
  },
  take: 50
});
